function renderOverlay(entry) { const emotionColors = { joy: '#FFD700', sorrow: '#708090', desire: '#FF4500', pride: '#8A2BE2', fear: '#2F4F4F', trust: '#32CD32' }; const roleIcons = { creator: '🎨', supporter: '🤝', analyst: '📊', ritualist: '🔥', scout: '🕵️' }; const color = emotionColors[entry.emotion] || '#999'; const icon = roleIcons[entry.role] || '🌀'; return `
${icon} ${entry.alias}
Emotion: ${entry.emotion}
Role: ${entry.role}
Signal: ${entry.signal}
Timestamp: ${entry.timestamp}
`; }
